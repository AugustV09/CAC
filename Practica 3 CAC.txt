    IMR EQU 21H

    ORG 40
    DW GESTOR
    
    ORG 1000H
MSJ DB "VAMOS LAS INTERRUPCIONES!"
SAL DB 10  ;LF

    ORG 3000H
GESTOR:
    PUSH BX
    PUSH AX
    MOV BX, OFFSET MSJ
    MOV AL, OFFSET SAL - OFFSET MSJ
    INT 7  ;IMPRIMO EL MENSAJE
    MOV BX,OFFSET SAL
    MOV AL,1
    INT 7  ;IMPRIMO EL SALTO DE LINEA
    MOV AL,20H
    OUT IMR-1,AL  ;PONGO 20H EN EOI
    POP AX
    POP BX
    IRET

    ORG 2000H
    CLI    ;INHABILITO LAS INTERRUPCIONES
    MOV AL,11111110b  ; ENMASCARO TODAS LAS INTERRUPCIONES MENOS LA DE F10
    OUT IMR,AL
    MOV AL,10
    OUT IMR+3,AL
    STI    ;HABILITO LAS INTERRUPCIONES

LAZO:
    JMP LAZO
    HLT
    END
